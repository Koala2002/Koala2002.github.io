int Binary_Search(vector<int>& arr,int l,int r,const int find_ele){
    //一開始傳進來的r為arr的size-1
    if(l>r)return -1;
    int mid=(l+r)/2;
    if(arr[mid]==find_ele)return mid;
    else if(arr[mid]<find_ele)return Binary_Search(arr,mid+1,r,find_ele);
    else return Binary_Search(arr,l,mid-1,find_ele);
}

int Binary_Search(vector<int>& arr,const int find_ele){
    int start=0,step=arr.size()-1;
    while(step>0){
        if(start+step<arr.size()&&arr[start+step]<=find_ele)start+=step;
        else step/=2;
    }
    
    return start;
}
