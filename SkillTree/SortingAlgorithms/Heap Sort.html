時間複雜度：O(n·log(n))
空間複雜度：O(1)

    
void adj(vector<int> &arr,int root,int n){
    int temp=arr[root],child=root*2+1;

    while(child<n){
        if(child+1<n&&arr[child+1]>arr[child])child+=1;
        if(temp>=arr[child])break;
        arr[(child-(child%2==0))/2]=arr[child];	 
        child=child*2+1
    }
    arr[(child-(child%2==0))/2]=temp;
}

void HeapSort(vector<int> &arr){
    for(int a=(arr.size()-1)/2;a>=0;a--)adj(arr,a,arr.size());
	
    for(int a=arr.size()-1;a>=0;a--){	
        swap(arr[0],arr[a]);
        adj(arr,0,a);
    }	
}
