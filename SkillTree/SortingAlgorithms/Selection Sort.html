時間複雜度：O(n2)
空間複雜度：O(1)

        
//由小排到大
void Selection_Sort(vector<int> &arr,int len){
    for(int a=0;a<len;a++){
        int min_ele=arr[a],pos=a;
        for(int b=a+1;b<len;b++){
            if(arr[b]<arr[a]){
                min_ele=arr[b];
                pos=b;
            }
        }
        swap(arr[a],arr[pos]);
    }
}
