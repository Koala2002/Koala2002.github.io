時間複雜度：
    • 平均 ⇒ O(n·log(n))
    • 最差 ⇒ O(n2)

空間複雜度：O(log(n))

                
void QuickSort(vector<int> &arr,int l,int r){
    if(l>=r)return;
    int key=arr[l];
    int lidx=l,ridx=r;
	
    while(lidx<ridx){
        while(arr[ridx]>=key&&lidx<ridx)ridx--;
        while(arr[lidx]<=key&&lidx<ridx)lidx++;
		
        swap(arr[lidx],arr[ridx]);
    }
    swap(arr[l],arr[lidx]);

    QuickSort(arr,l,lidx-1);
    QuickSort(arr,lidx+1,r);
}
